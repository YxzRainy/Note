# 配置交换机端口安全

进全局模式



interface f 0/1

switchport mode access

switch port-security

exit



interface f 0/2

switchport mode access

switch port-security

exit



interface f 0/1

switchport port-security maximum 1

exit



interface f 0/2

switchport port-security maximum 1

exit



interface f 0/1

switchport port-security mac-address sticky

no shutdown

exit



interface f 0/2

switchport port-security mac-address sticky

no shutdown

exit



interface range f 0/1-10

no shutdown

exit



interface f 0/1

switchport port-security violation restrict

exit



interface f 0/2

switchport port-security violation restrict

exit



interface range f 0/1-10

shutdown

exit



interface range f 0/10-19

shutdown

exit



interface range f 0/20-24

shutdown

exit
