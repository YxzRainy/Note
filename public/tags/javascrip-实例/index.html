JavaScript 自动阅读

自动阅读 JavaScrip var start = document.getElementsByTagName('div')[0]; var stop = document.getElementsByTagName('div')[0]; var time; // 定义一个 key，用来判断自动阅读是否暂停 // 如果不加判断，setInterval() 就会被重复执行多个（导致自动阅读的速度增加），而 clearInterval() 则只能暂停第一个 setInterval()（导致自动阅读无法被暂停） var key = '自动阅读已暂停'; start.onclick = function () { // 判断自动阅读的状态 if (key == '自动阅读已暂停') { // 用 setInterval() 反复执行 scrollBy() 实现页面自动向下滚动 time = setInterval(function () { window.scrollBy(0, 10); }, 100); // 自动阅读已开始 key = '自动阅读已开始'; } }; // 自动阅读已暂停 stop.onclick = function () { clearInterval(time); key = '自动阅读已暂停'; };
