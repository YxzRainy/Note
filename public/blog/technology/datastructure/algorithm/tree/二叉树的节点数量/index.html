二叉树的节点数量#

递归法#

  1. 判断根节点是否有左右子节点，有则递归调用NodeNum()，则子节点 + 1。
  2. 左右子节点数量加上根节点数量。
·include <stdio.h>
#include <stdlib.h>

// 定义节点
typedef struct node
{
    int data;
    struct node *lchild, *rchild;
} n;

// 创建一个节点
node * get_a_node(int data)
{
    node *c = (node *)malloc(sizeof(node));
    c->data = data;
    c->lchild = 0;
    c->rchild = 0;
    return c;
}


// *p：根节点
int NodeNum(node *p)
{
    // 节点总数
    int nums;

    // 左节点数量
    int left = 0;

    // 右节点数量
    int right = 0;

    // 有左节点
    if (p->lchild)
    {
        // 左节点 + 1
        left = NodeNum(p->lchild);
    }
    // 有右节点
    if (p->rchild)
    {
        // 右节点 + 1
        right = NodeNum(p->rchild);
    }
    // 左右节点数量加上根节点数量
    // 这个 + 1 很重要，因为每次调用 NodeNum()，就说明有节点，则 nums + 1。
    nums = left + right + 1;
    return nums;
}


int main(void)
{
    // 构建二叉树
    node *t = get_a_node(1);
    t->lchild = get_a_node(2);
    t->rchild = get_a_node(3);
    t->lchild->rchild = get_a_node(5);
    t->rchild->rchild = get_a_node(7);
    t->rchild->rchild->lchild = get_a_node(14);

    int num = NodeNum(t);
    printf("二叉树的结点个数为：%d\n",num);
    return 0;
}
