哈夫曼树#

#include <stdio.h>
#include <stdlib.h>
#include <queue>
using namespace std;

typedef struct node{
	int data;
	struct node *lchild, *rchild;
} n;

//创建一个节点
node * get_a_node(int data){
	node *c = (node *)malloc(sizeof(node));
	c->data = data;
	c->lchild = 0;
	c->rchild = 0;
	return c;
}


int main(){
	queue<node *>q;
	//构建一棵树
	node *t = get_a_node(1);
	t->lchild = get_a_node(2);
	t->rchild = get_a_node(3);
	t->lchild->rchild = get_a_node(5);
	t->rchild->rchild = get_a_node(7);
	t->rchild->rchild->lchild = get_a_node(14);

	q.push(t);
	//在这里测试你所完成的函数
	while (q.size() > 0){
		node *c = q.front();
		if(c->lchild)
			q.push(c->lchild);
		if(c->rchild)
			q.push(c->rchild);
		printf("%d ",c->data);
		q.pop();
	}
}
