二分查找#

#include <stdio.h>
void main(){
	int arr[] ={1,3,7,10,12,14,17,20,21,23};
	int num = 50;
	int left = 0, right = 10;
	int flag = 0;
	while(left < right){
		int middle = (left + right)/2;
		if(arr[middle] > num){
			right = middle;
		}
		if(arr[middle] < num){
			left = middle +1;
		}
		if(arr[middle] == num){
			printf("%d",middle);
		    flag = 1;
			break;
		}
	}
	if(flag == 0 ){
		printf("not found\n");
	}
}
