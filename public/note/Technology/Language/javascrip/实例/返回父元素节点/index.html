返回父元素节点.#

JavaScrip#

function re(e, n) {
	// 如果 e 不为 null 并且 n 不为 0
	while (e && n) {
		// 获取当前节点的父元素
		e = e.parentElement;
		// 每次循环一次，则想上找一层的父元素，直到 n 为 0
		n--;
	}
	return e;
}

var i = document.getElementsByTagName('i')[0];

var result = re(i, 3);
console.log(result);
