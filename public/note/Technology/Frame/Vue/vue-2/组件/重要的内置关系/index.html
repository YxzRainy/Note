重要的内置关系#

VueConponent.prototype.__proto__ === Vue.prototype

为什么要有这个关系？

让该组件实例对象可以访问到 Vue 原型上的属性和方法。

HTML

<div id="app"></div>

JavaScrip

// VueComponent 的实例对象 school
const school = {
	template: `
    <div>
    <h1>{{schoolName}}</h1>
    <button @click='fun'>点击</button>
    </div>
    `,
	data() {
		return {
			schoolName: '哈佛',
		};
	},
	methods: {
		fun() {
			// 调用 school 上没有的 val
			console.log(this.val);
		},
	},
};
Vue.component('school', school);

// Vue
new Vue({
	template: `<school></school>`,
	el: '#app',
	data: {},
});
// Vue 的 prototype
Vue.prototype.val = '我是 Vue 原型对象上的属性';
Vue.config.productionTip = false;
