printf

用于将变量的内容输出到显示器.

直接输出

1printf("字符串");

print(“输出控制符”,参数)

1int i = 10;//将数值 10 的二进制代码储存到变量 i 中 。
2printf("%o", a);//将变量 i（数值 10） 的二进制代码以八进制输出，a 为参数。

printf(“输出控制符1 输出控制符2”,参数1,参数2)

1 int a = 10;
2 int i = 15;
3 printf("%d %d\n", a, i);//控制符和参数必须一一对应。

printf(“输出控制符 非输出控制符”,参数)

1int a = 10;//将数值 10 的二进制代码储存到变量 i 中 。
2printf("%o\n",a);//将变量 i（数值 10 的二进制代码以八进制输出；\n 为非输出控制符。
