选择控制

某些代码可能执行，也可能不执行，有选择的执行某些代码。

if

在程序当中,并不是所有的语句都是自上而下依次的执行，有时候会依据条件来选择性的执行某段代码，这个时候就需要用到 if 语句。

如果 1 为真，则执行 A；如果 1 为假，则不执行 A。

1#include <stdio.h>
2int main(void) {
3	if (1)
4	        A；
5}

1#include <stdio.h>
2int main(void) {
3	if (2 > 1)
4	        printf("if 语句为 true\n");
5}

if 语句默认只能控制 A，不能控制B（即无论 1 为 true 还是 false， B 都会执行）。

1#include <stdio.h>
2int main(void) {
3	if (1)
4	        A;
5	B;
6}

1#include <stdio.h>
2int main(void) {
3	if (2 > 1)
4	        printf("被 if 语句控制\n");
5	printf("不被 if 语句控制\n");
6}

if 语句默认只能控制一个 A，如果想控制多个语句（A 和 B），就必须将 A 和 B 用 {} 括起来。

1#include <stdio.h>
2int main(void) {
3	if(1) {
4		A;
5		B;
6	}
7}

1#include <stdio.h>
2int main(void) {
3	if (2 > 1) {
4		printf("被 if 语句控制\n");
5		printf("被 if 语句控制\n");
6	}
7}

if else

if else 语句会判定 1 的真假，表达式为 true 则执行 A 和 B；1 为 false 则执行 C；D为必须执行的语句（D 不受 if 语句控制）。

1#include <stdio.h>
2int main(void) {
3	if (1) {
4		A;
5		B;
6	} else
7	        C;
8	D;
9}

1#include <stdio.h>
2int main(void) {
3	if (4 > 2) {
4		printf("正确\n");
5		printf("二次确认也正确\n");
6	} else
7	        printf("不正确\n");
8	printf("必定会执行\n");
9}

if else 连续

使用 if else 语句连续判断1、2、4，若某个表达式为 true 则返回其对应的A、B、C、D，若1、2、3 全部为 false，则返回 D。

1#include <stdio.h>
2int main(void) {
3	if (1)
4	        printf("A\n"); else if (2)
5	        printf("B\n"); else if (3)
6	        printf("C\n"); else
7	        printf("D\n");
8}

1#include <stdio.h>
2int main(void) {
3	if (4 == 2)
4	        printf("表达式一为假\n"); else if (2 == 5)
5	        printf("表达式二也为假\n"); else if (3 == 7)
6	        printf("表达式三还是为假\n"); else
7	        printf("三个表达式全部为假\n");
8}

if else 被分割

 1#include <stdio.h>
 2int main(void) {
 3	if (4 == 2)
 4	        printf("不正确\n"); else if (2 == 5)
 5	        printf("也不正确\n");
 6	printf("程序会因为该行报错\n");
 7	//该行将原本完整的 if else 语句分割了，导致下面的 if 语句不完整。 else if (3 == 7)
 8	        printf("还是不正确\n"); else
 9	        printf("全部错误\n");
10}

互换两个值

 1#include <stdio.h>
 2int main(void) {
 3	int i = 3;
 4	int j = 5;
 5	int t;
 6	//定义临时的变量 t。
 7	t = i;
 8	//将变量 i 的值赋给变量 t，作为变量 i 值的备份。
 9	i = j;
10	//将变量 j 的值赋给变量 i，这时的变量 i 和变量 t 的值相同。
11	j = t;
12	//将变量 t 中的值，也就是 i 的值赋给变量 j。
13	printf("i = %d\nj = %d\n", i, j);
14}

返回最大值

 1#include <stdio.h>
 2int main(void) {
 3	int a, b, c;
 4	printf("请输入三个数：");
 5	scanf("%d %d %d", &a, &b, &c);
 6	if (a == b || b == c || c == a)
 7	        printf("请勿输入相同的值！\n"); else {
 8		if (a > b && a > c)
 9		            printf("a 最大!\n"); else if (b > a && b > c)
10		            printf("b 最大!\n"); else if (c > a && c > b)
11		            printf("c 最大!\n");
12	}
13}

降序

 1#include <stdio.h>
 2int main(void) {
 3	int a, b, c, t;
 4	printf("请输入三个值：\n");
 5	scanf("%d %d %d", &a, &b, &c);
 6	if (b > a) {
 7		t = a;
 8		a = b;
 9		// 将 b 赋给 a，使 b 代替 a 最大的位置
10		b = t;
11		//使 a 代替 b 中间的位置
12	}
13	if (c > a) {
14		t = a;
15		a = c;
16		// 将 c 赋给 a，使 c 代替 a 最大的位置
17		c = t;
18		// 使 a 代替 c 最小的位置
19	}
20	if (c > b) {
21		t = b;
22		b = c;
23		//将 c 赋给 b，使 c 代替 b 中间的位置
24		c = t;
25		//使 b 代替 c 最小的位置
26	}
27	printf("最大值：%d\n中间值：%d\n最小值：%d\n", a, b, c);
28	//默认为 a 最大，b 次之，c 最小；比较三个值大小使得 a、b、c，交换最大、中间、最小的位置。
29}

switch

switch是选择不是循环，如果在switch 中出现了 berak。该语句的功能只是退出 switch，并继续执行它下 面的语句。在 switch 中出现continue 是错误的，除非 switch 本身就属于 for 或 while 循环的一部分。
