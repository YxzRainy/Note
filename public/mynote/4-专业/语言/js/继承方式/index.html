继承方式

原型链继承

缺点

会使一个对象过多继承一些没用或不需要的属性，则会导致代码的执行效率大大降低。反对反对反对法大幅度发反对发的

 1Grand.prototype.blue = 'blue';
 2function Grand() {}
 3
 4var grandfather = new Grand();
 5
 6Father.prototype = grandfather;
 7
 8function Father() {
 9	this.red = 'red';
10	this.money = {
11		card: 'pencil',
12	};
13	this.value = '100';
14}
15
16var father = new Father();
17
18Son.prototype = father;
19
20function Son() {
21	this.orange = 'orange';
22}
23var son = new Son();
24
25console.log(son.orange);
26// 返回 orange。
27console.log(son.red);
28// 返回 red，虽然对象 son 本身没有属性 red，但它的原型 father 上有属性 red。
29console.log(son.blue);
30// 返回 blue，虽然对象 son 本身没有属性 blue，其原型 father 也没有这个属性，
31// 但原型 father 的原型 grand 上有属性 blue。
32son.money.box = 'bear';
33// 通过调用对象 son 给原型对象 father 的属性 money（money 的值必须是引用值）增加一个引用值 box。
34console.log(son.money.box);
35// 返回 bear。
36son.value++;
37son.value++;
38// 先将原型对象 father 的属性 value 拷贝给自身（son），然后拿着拷贝过来的属性 value 进行自增。
39// 因此这里的自增并不会影响原型对象 father 的属性 value。
40console.log(son.value);
41// 返回 102，因为自增了两次。
42console.log(father.value);
43// 因为原型对象 father 将其自身的属性 value 拷贝给了对象 son，
44// 然后对象 son 拿着拷贝过来的属性 value 进行的自增，
45// 因此对象 father 的属性 value 并没有发生改变。

借用构造函数继承

使用 call 与 apply 实现继承（伪造对象），即借用其他构造函数的属性。

有需求时可以使用该方法实现继承。

缺点

不能借用其构造函数的原型（不算继承，只是借用）。

视觉上省去了部分代码，但调用对象时，该对象的构造函数则会多调用一个或多个构造函数，因此也会使代码的执行效率大大降低。

详见 Function

共享原型继承

将一个函数的原型指向共享给另一个函数的原型，使这个两个原型都指向同一个原型（即它们都继承自同一个原型），即这两个函数的原型共用一个原型。

缺点

不能单独给其中一个函数的原型添加私有属性，即添加一个函数的原型的属性时，另一个函数的原型也可以调用这个属性。

 1Demo.prototype.name = 'windy';
 2Demo.prototype.age = 18;
 3// 函数 Demo 与函数 Test 的共有原型对象 Demo.prototype。
 4function Demo() {
 5	this.name = 'tiger';
 6}
 7
 8function Test() {}
 9
10function inherit(a, b) {
11	a.prototype = b.prototype;
12	// 用函数 inherit 封装一个有继承功能的函数，相当于 Test.prototype = Demo.prototype ，
13	// 也就是将函数 Demo 的原型地址赋给函数 Test，即将函数 Demo 的原型共享给函数 Test ,
14	// 使这两个函数共用一个原型。
15}
16
17inherit(Test, Demo);
18
19Test.prototype.male = 'boy';
20// 单独给函数 Test 的原型添加属性 male，但因为该函数与函数 Demo 的原型指向都指向同一个原型，
21// 因此会导致这个属性 male 依旧为这两个函数的共有属性。
22var demo = new Demo();
23var test = new Test();
24
25console.log(demo.name);
26// 返回 tiger。
27console.log(test.age);
28//返回 18。
29console.log(demo.male);
30// 返回 boy。
31console.log(test.male);
32// 返回 boy。
33// 因为这两个对象的原型指向都指向同一个原型对象，因此调用这两个对象的属性 male 都会返回 body。

圣杯模式继承

用来继承已有原型对象（A.prototype）中的成员（主要是公用方法），同时根据自己的需求修改原型对象（A.prototype），以定制符合我们要求的构造函数 B，这个修改对已有的实例（a1，a2，…）不会产生影响。

基本没有缺点

 1function inherit(a, b) {
 2	function Rain() {}
 3	// 用函数 Rain 作为中间层
 4	Rain.prototype = b.prototype;
 5	// 相当于将函数 Demo 的原型地址共享给函数 Rain 的原型，
 6	// 也就是将函数 Demo 的原型共享给函数 Rain 的原型。使这两个函数的原型共用一个原型
 7	a.prototype = new Rain();
 8	// 使用原型链形成了继承关系，现在改 Test.prototype 就不会影响 Demo.prototype
 9	a.prototype.constructor = a;
10	//  默认对象 test 的属性 construction 的指向为函数 Demo，
11	// 这里将对象 test 的属性 construction 的指向改为 test 本身。
12	a.prototype.uber = b.prototype;
13	// 使构造出的对象 test 可以寻找到自己的超类（超级父级），
14	// 即可以知道该对象真正继承的是哪一个函数
15}
16
17// 但建议使用这种写法：
18var inherits = (function () {
19	function Rain() {}
20	// 函数 Rain 为私有化变量，用作于中间层
21	return function (a, b) {
22		Rain.prototype = b.prototype;
23		a.prototype = new Rain();
24		a.prototype.constructor = a;
25		a.prototype.uber = b.prototype;
26	};
27})();
28
29Demo.prototype.name = 'windy';
30Demo.prototype.age = 18;
31
32function Demo() {
33	this.name = 'tiger';
34}
35
36function Test() {}
37
38inherit(Test, Demo);
39
40Test.prototype.male = 'boy';
41// 给函数 Test 添加属性 male，则属性 male 是该函数的私有属性，
42// 即该函数的属性 male 不会共享给函数 Demo。
43var test = new Test();
44var demo = new Demo();
45
46console.log(test.male);
47// 返回，boy。
48console.log(demo.male);
49//返回 undefined，调用对象 demo 的属性 male，但因为该属性没有被对象 test 共享给对象 demo，
50// 因此这里相当于调用一个对象没有的属性。
