23    public static void bubbleSort(int[] arr) {
24        //  arr.length - 1，即 5 - 1 = 4，
25        // 这里的循环控制的是总共进行 n 轮排序，这里只进行四轮排序
26        // 为什么只进行四轮，因为通过推算，进行 4 轮排序后就可以得出最终结果。
27        for (int i = 0; i < arr.length - 1; i++) {
28            /*
29            第一轮排序开始，外层变量 i = 0，再 5 - 0 - 1 = 4，意味着要从左往右对调 4 次。
30            6 4 8 2 9
31            6 4        第 1 次对调,j = 0
32              8 4      第 2 次对调,j = 1
33                4 2    第 3 次对调,j = 2
34                  9 2  第 4 次对调,j = 3
35            6 8 4 9 2, 第一轮对调完毕,对调了四次。
36            此时的 j = 4,不满足循环条件,循环结束,退出循环,开始进行第 2 轮排序.
37
38            第二轮排序开始,外层变量 i = 1，再 5 - 1 - 1 = 3 意味着要从左往右对调 3 次。
39            将 j 重新赋值为 0 (j 原为 4),开始内层循环,这次依旧是循环四次
40            6 8 4 9 2
41            8 6        对调 1 次，j = 0
42              6 4      对调 2 次，j = 1
43                9 4    对调 3 此，j = 2
44                    2  剩下的数 2 已经确定了
45            8 6 9 4 2,第二轮对调完毕,此时的 j = 4,不满足循环条件,循环结束,退出循环,开始进行第 2 轮排序。
46
47            第三轮排序开始，外层变量 i = 2，再 5 - 2 - 1 = 2，意味着要从左往右对调 2 次。
48            8 6 9 4 2
49            8 6 对调 1 次
50              9 6 对调 2 次
51                  4 2，剩下的数 4、2 已经确定了
52            8 9 6 4 2，第三轮对调完毕,只对调了两次
53
54            第四轮排序开始，外层变量 i = 3，再 5 - 3 - 1 = 1，意味着要从左往右对调 1 次。
55            8 9 6 4 2
56            9 8，对调 1 次
57                6 4 2 剩下的数 6、4、2 已经确定了。
58            9 8 6 4 2，第四轮对调完毕，只对调了一次。
59            */
60
61            for (int j = 0; j < arr.length - i - 1; j++) {
62                //两个数比较，如果前者小于后者，则元素从大到小进行排序，
63                // 反之前者大于后者，则元素从小到大进行排序
64                if (arr[j] > arr[j + 1]) {
65                    //定义一个临时变量 temp，用于后面对调数组中两个数的位置
66                    // 将前者赋给 temp
67                    int temp = arr[j];
68                    //将后者赋给前者
69                    arr[j] = arr[j + 1];
70                    //再将临时变量(前者)赋给后者,最终实现了前者与后者的位置互换,此时的 temp 的值是前者.
71                    arr[j + 1] = temp;
72                }
73            }
74            //外层循环每执行完毕一次，代表着一轮的排序结束，用于计算排序进行了多少次
75            System.out.print("第" + (i + 1) + "轮排序后:");
76            //返回最总的排序结果
77            printArray(arr);
78        }
79    }
