avaScript部分

接下来就是重头戏了，也是实现搜索功能的核心部分，搜索逻辑的实现；

再来大致分析一下逻辑，和实现的思路：

  1. 利用XMLHttpRequest()获取站内feed.xml内的所有数据，保存到一个XML DOM对象中；
  2. 将XML对象中的文章标题、链接、内容、索引等通过getElementsByTagName()等方法获取并保存到对应数组变量中；
  3. 用户在输入框输入查找内容，提交后内容保存到一个字符串类型变量中；
  4. 遍历保存文章内容的数组，通过.search()等方法和输入值进行匹配；
  5. 匹配成功后得到所有匹配成功的数组元素的索引值，该索引值也是该内容的标题、链接数组对应的索引值；
  6. 将最终搜集的文章标题、链接，以及匹配到的内容片段摘取输出到页面；

这里附上最终的 js 实现代码与注释：

  1// 获取搜索框、搜索按钮、清空搜索、结果输出对应的元素
  2var searchBtn = document.querySelector('.search-start');
  3var searchClear = document.querySelector('.search-clear');
  4var searchInput = document.querySelector('.search-input');
  5var searchResults = document.querySelector('.search-results');
  6
  7// 申明保存文章的标题、链接、内容的数组变量
  8var searchValue = '',
  9    arrItems = [],
 10    arrContents = [],
 11    arrLinks = [],
 12    arrTitles = [],
 13    arrResults = [],
 14    indexItem = [],
 15    itemLength = 0;
 16var tmpDiv = document.createElement('div');
 17tmpDiv.className = 'result-item';
 18
 19// ajax 的兼容写法
 20var xhr = new XMLHttpRequest() || new ActiveXObject('Microsoft.XMLHTTP');
 21xhr.onreadystatechange = function () {
 22    if (xhr.readyState == 4 && xhr.status == 200) {
 23        xml = xhr.responseXML;
 24        arrItems = xml.getElementsByTagName('item');
 25        itemLength = arrItems.length;
 26        
 27        // 遍历并保存所有文章对应的标题、链接、内容到对应的数组中
 28        // 同时过滤掉 HTML 标签
 29        for (i = 0; i < itemLength; i++) {
 30            arrContents[i] = arrItems[i].getElementsByTagName('description')[0].
 31                childNodes[0].nodeValue.replace(/<.*?>/g, '');
 32            arrLinks[i] = arrItems[i].getElementsByTagName('link')[0].
 33                childNodes[0].nodeValue.replace(/<.*?>/g, '');
 34            arrTitles[i] = arrItems[i].getElementsByTagName('title')[0].
 35                childNodes[0].nodeValue.replace(/<.*?>/g, '');
 36        }
 37    }
 38}
 39
 40// 开始获取根目录下 feed.xml 文件内的数据
 41xhr.open('get', '/feed.xml', true);
 42xhr.send();
 43
 44searchBtn.onclick = searchConfirm;
 45
 46// 清空按钮点击函数
 47searchClear.onclick = function(){
 48    searchInput.value = '';
 49    searchResults.style.display = 'none';
 50    searchClear.style.display = 'none';
 51}
 52
 53// 输入框内容变化后就开始匹配，可以不用点按钮
 54// 经测试，onkeydown, onchange 等方法效果不太理想，
 55// 存在输入延迟等问题，最后发现触发 input 事件最理想，
 56// 并且可以处理中文输入法拼写的变化
 57searchInput.oninput = function () {
 58    setTimeout(searchConfirm, 0);
 59}
 60//输入框获取焦点的时间
 61searchInput.onfocus = function () {
 62    searchResults.style.display = 'block';
 63}
 64
 65function searchConfirm() {
 66    if (searchInput.value == '') {
 67        searchResults.style.display = 'none';
 68        searchClear.style.display = 'none';
 69    } else if (searchInput.value.search(/^\s+$/) >= 0) {
 70        // 检测输入值全是空白的情况
 71        searchInit();
 72        var itemDiv = tmpDiv.cloneNode(true);
 73        itemDiv.innerText = '请输入有效内容...';
 74        searchResults.appendChild(itemDiv);
 75    } else {
 76        // 合法输入值的情况
 77        searchInit();
 78        searchValue = searchInput.value;
 79        // 在标题、内容中查找
 80        searchMatching(arrTitles, arrContents, searchValue);
 81    }
 82}
 83
 84// 每次搜索完成后的初始化
 85function searchInit() {
 86    arrResults = [];
 87    indexItem = [];
 88    searchResults.innerHTML = '';
 89    searchResults.style.display = 'block';
 90    searchClear.style.display = 'block';
 91}
 92
 93function searchMatching(arr1, arr2, input) {
 94    // 忽略输入大小写
 95    input = new RegExp(input, 'i');
 96    // 在所有文章标题、内容中匹配查询值
 97    for (i = 0; i < itemLength; i++) {
 98        if (arr1[i].search(input) !== -1 || arr2[i].search(input) !== -1) {
 99            // 优先搜索标题
100            if (arr1[i].search(input) !== -1) {
101                var arr = arr1;
102            } else {
103                var arr = arr2;
104            }
105            indexItem.push(i);  // 保存匹配值的索引
106            var indexContent = arr[i].search(input);
107            // 此时 input 为 RegExp 格式 /input/i，转换为原 input 字符串长度
108            var l = input.toString().length - 3;
109            var step = 10;
110            
111            // 将匹配到内容的地方进行黄色标记，并包括周围一定数量的文本
112            arrResults.push(arr[i].slice(indexContent - step, indexContent) +
113                '<mark>' + arr[i].slice(indexContent, indexContent + l) + '</mark>' +
114                arr[i].slice(indexContent + l, indexContent + l + step));
115        }
116    }
117
118    // 输出总共匹配到的数目
119    var totalDiv = tmpDiv.cloneNode(true);
120    totalDiv.innerHTML = '总匹配：<b>' + indexItem.length + '</b> 项';
121    searchResults.appendChild(totalDiv);
122
123    // 未匹配到内容的情况
124    if (indexItem.length == 0) {
125        var itemDiv = tmpDiv.cloneNode(true);
126        itemDiv.innerText = '未匹配到内容...';
127        searchResults.appendChild(itemDiv);
128    }
129
130    // 将所有匹配内容进行组合
131    for (i = 0; i < arrResults.length; i++) {
132        var itemDiv = tmpDiv.cloneNode(true);
133        itemDiv.innerHTML = '<b>《' + arrTitles[indexItem[i]] +
134            '》</b><hr />' + arrResults[i];
135        itemDiv.setAttribute('onclick', 'changeHref(arrLinks[indexItem[' + i + ']])');
136        searchResults.appendChild(itemDiv);
137    }
138}
139
140function changeHref(href) {
141
142    // 在当前页面点开链接的情况
143    location.href = href;
144
145    // 在新标签页面打开链接的代码，与上面二者只能取一个，自行决定
146    // window.open(href);
147}
