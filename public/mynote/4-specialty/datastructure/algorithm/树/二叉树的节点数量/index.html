二叉树的节点数量

二叉树的节点数量#

#include <stdio.h>
#include <stdlib.h>

typedef struct node
{
    int data;
    struct node *lchild, *rchild;
} n;

// 创建一个节点
node * get_a_node(int data)
{
    node *c = (node *)malloc(sizeof(node));
    c->data = data;
    c->lchild = 0;
    c->rchild = 0;
    return c;
}


// 求树的节点数量
int node_num(node *p)
{
    int left = 0;
    int right = 0;
    if (p->lchild)
    {
        left = node_num(p->lchild);
    }
    if (p->rchild)
    {
        right = node_num(p->rchild);
    }
    int nums = left + right +1;
    return nums;
}


int main(void)
{
    //构建一棵树
    node *t = get_a_node(1);
    t->lchild = get_a_node(2);
    t->rchild = get_a_node(3);
    t->lchild->rchild = get_a_node(5);
    t->rchild->rchild = get_a_node(7);
    t->rchild->rchild->lchild = get_a_node(14);

    // 树的结点个数
    int num = node_num(t);
    printf("树的结点个数为：%d\n",num);

    return 0;
}
