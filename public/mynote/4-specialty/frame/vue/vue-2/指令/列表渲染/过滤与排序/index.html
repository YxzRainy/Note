过滤与排序#

当用computed和watch都能实现某一个功能的时候，优先使用computed。

过滤#

分别用两种方式实现过滤，computed 与watch，推荐使用前者。

HTML

<div id="app">
    <h1>模糊搜索</h1>
    <input type="text" name="" id="" placeholder="请输入名字" v-model="keyWord">
    <ul>
        <li v-for="p in filterPersons" :key="p.id">
            {{p.name}}-{{p.age}}-{{p.sex}}
        </li>
    </ul>

</div>

JavaScrip

watch 实现

var vm = new Vue({
	el: '#app',
	data: {
		keyWord: '',
		persons: [
			{ id: '001', name: '马冬梅', age: 20, sex: '女' },
			{ id: '002', name: '周冬雨', age: 21, sex: '女' },
			{ id: '003', name: '周杰伦 ', age: 22, sex: '男' },
			{ id: '004', name: '温兆伦 ', age: 18, sex: '男' },
		],
		// 用来接受过滤后的结果
		filterPersons: [],
	},
	methods: {},
	watch: {
		keyWord: {
			// 初始化 filterPersons 数组
			immediate: true,
			handler(val) {
				// 将过滤后的结果赋给 persons
				this.filterPersons = this.persons.filter((p) => {
					// 返回查找到的字符串第一次出现的索引
					return p.name.indexOf(val) !== -1;
				});
			},
		},
	},
});

computed 实现

var vm = new Vue({
	el: '#app',
	data: {
		keyWord: '',
		persons: [
			{ id: '001', name: '马冬梅', age: 20, sex: '女' },
			{ id: '002', name: '周冬雨', age: 21, sex: '女' },
			{ id: '003', name: '周杰伦 ', age: 22, sex: '男' },
			{ id: '004', name: '温兆伦 ', age: 18, sex: '男' },
		],
	},
	computed: {
		filterPersons() {
			return this.persons.filter((p) => {
				return p.name.indexOf(this.keyWord) !== -1;
			});
		},
	},
});

排序#

用computed实现

HTML

<!DOCTYPE html>
<html lang="en">

    <head>
        <meta charset="UTF-8" />
        <meta http-equiv="X-UA-Compatible" content="IE=edge" />
        <meta name="viewport" content="width=device-width, initial-scale=1.0" />
        <script src="https://cdn.bootcdn.net/ajax/libs/vue/2.6.9/vue.js"></script>
        <title>模糊搜索</title>
        <style>

        </style>
    </head>

    <body>
        <div id="app">
            <h1>模糊搜索</h1>
            <input type="text" name="" id="" placeholder="请输入名字" v-model="keyWord">
            <button @click="sortType ='2'">年龄升序</button>
            <button @click="sortType ='1'">年龄降序</button>
            <button @click="sortType ='0'">原顺序</button>

            <ul>
                <li v-for="p in filterPersons" :key="p.id">
                    {{p.name}}-{{p.age}}-{{p.sex}}
                </li>
            </ul>

        </div>
    </body>
    <script src="./js/main.js"></script>

</html>

JavaScript

var vm = new Vue({
	el: '#app',
	data: {
		keyWord: '',
		// 0 原顺序、1年龄降序、2 年龄升序
		sortType: '0',
		persons: [
			{ id: '001', name: '马冬梅', age: 28, sex: '女' },
			{ id: '002', name: '周冬雨', age: 21, sex: '女' },
			{ id: '003', name: '周杰伦', age: 52, sex: '男' },
			{ id: '004', name: '温兆伦', age: 31, sex: '男' },
		],
	},
	computed: {
		filterPersons() {
			// 保存过滤后的数组，以便后续排序
			var filterArr = this.persons.filter((p) => {
				// 检测数组中是否含有输入框中的文本
				return p.name.indexOf(this.keyWord) !== -1;
			});
			// 判断如何进行排序
			// 年龄降序
			if (this.sortType == '1') {
				filterArr.sort((p1, p2) => {
					return p2.age - p1.age;
				});
				// 年龄升序
			} else if (this.sortType == '2') {
				filterArr.sort((p1, p2) => {
					return p1.age - p2.age;
				});
			}
			return filterArr;
		},
	},
});
