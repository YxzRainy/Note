keyframes

@keyframes rotate {
	/* 从 0° 开始旋转 */
	from {
		transform: translate(-50%, -50%) rotate(0deg);
	}
	/* 旋转到 360 ° */
	to {
		transform: translate(-50%, -50%) rotate(360deg);
	}
}
